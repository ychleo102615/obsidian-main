throttle
```ts
function throttle(fn: () => void, interval: number): () => void {
	let lastTime = 0
	return function(...args) {
		const now = new Date()
		if (now - lastTime > interval) {
			fn.apply(this, args)
			lastTime = now
		}
	}
}
```
