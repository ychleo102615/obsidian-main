throttle
```ts
function throttle<T extends (...args: any[]) => any>(
	fn: T,
	interval: number
): (...args: Parameters<T>) => void {
	let lastTime = 0
	return function(this: unkonwn, ...args: Parameters<T>) {
		const now = Date.now()
		if (now - lastTime > interval) {
			fn.apply(this, args)
			lastTime = now
		}
	}
}
```
