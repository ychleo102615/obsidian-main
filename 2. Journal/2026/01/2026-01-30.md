throttle
```ts
function throttle<T extends (...args: any[]) => any>(
	fn: T,
	interval: nu
): () => void {
	let lastTime = 0
	return function(...args) {
		const now = new Date()
		if (now - lastTime > interval) {
			fn.apply(this, args)
			lastTime = now
		}
	}
}
```
