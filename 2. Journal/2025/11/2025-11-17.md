題目1
```js
productInventory = {
	item_sku_A1001: 150,
	item_sku_C5470: 300,
	item_sku_D0089: 75
}
  
orderedProductIds = [
	'item_sku_C5470',
	'item_sku_A1001',
	'item_sku_D0089'
]
  
sortInventory(productInventory, orderedProductIds)

/*
[
	{id: 'item_sku_C5470', amount: 300},
	{id: 'item_sku_A1001', amount: 150},
	{id: 'item_sku_D0089', amount: 75}
]
*/
```

ans
```js
function sortInventory(productInventory, orderedProductIds) {
	// my version
	const res = []
	for (let i = 0; i < orderedProductIds.length; i++) {
	const id = orderedProductIds[i]
		res.push({
			id: id,
			amount: productInventory[id]
		})
	}
	return res
	
	// feedback better version
	return orderedProductIds.map(id => new Object({
		id: id,
		amount: productInventory[id]
	}));
}

const res = sortInventory(productInventory, orderedProductIds)
console.log(res)
```
