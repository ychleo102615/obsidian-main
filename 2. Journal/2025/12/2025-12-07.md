## 每日精華

[code challenge](https://www.freecodecamp.org/learn/daily-coding-challenge/2025-12-06)

```js
function getMonthNumberFromName(monthName) {
	const m = new Date(`${monthName} 1, 2022`).getMonth() + 1;
	return m.toString().padStart(2, "0")
}

function formatDate(dateString) {
	const [m, d, y] = Array.from(dateString.matchAll(/\w+/g), x => x[0])
	return `${y}-${getMonthNumberFromName(m)}-${d.padStart(2,"0")}`
}

console.log(formatDate("December 6, 2025"))
console.log(formatDate("January 1, 2000"))
console.log(formatDate("November 11, 1111"))
console.log(formatDate("February 29, 1992"))
```
