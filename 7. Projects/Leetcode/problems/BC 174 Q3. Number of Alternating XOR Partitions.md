在解題目的時候，我自己想到的是 DP 的方法：
```ts
function alternatingXOR(nums: number[], target1: number, target2: number): number {
    const dp = {}
    const n = nums.length
    const dfs = (i: number, step: number): number => {
        if (i === n) {
            return 1
        }
        const key = [i,step].join(",")
        if (key in dp) {
            return dp[key]
        }
        let res = 0
        let count = 0
        const target = (step === 0) ? target1 : target2
        const nextStep = (step + 1) % 2
        for (let j = i; j < n; j++) {
            res ^= nums[j]
            if (res === target) {
                count = (count + dfs(j+1, nextStep)) % (1000000007)
            }
        }
        dp[key] = count
        return count
    }
    return dfs(0, 0) 
};
```

而且我還沒注意到這題目叫你用 MOD $1e9+7$。
